# Browser video 2× slow-mo (Phase 4-lite) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A browser page that takes a short clip, precomputes the interpolated in-between frames, and plays the doubled sequence back at the original fps — smooth 2× slow-motion.

**Architecture:** Extract the ort-web session + single-pair interpolation into `web/rife_session.js` (shared with `demo.html`). `web/slowmo.html` samples N frames from a `<video>` via seek→canvas, calls `interpolate` per consecutive pair, interleaves via the pure `assembleSlowmo` (in `web/slowmo.js`), and plays the sequence on a canvas with a rAF loop.

**Tech Stack:** ONNX Runtime Web (WebGPU/wasm), vanilla JS ES modules, `<video>`+canvas frame sampling, node 22 (unit test), Playwright (E2E).

## Global Constraints

- Reuse `web/rife_prepost.js` (BGR/÷255/pad/crop) and the fixed engine size `EW=1280, EH=736`; model `assets/rife_lite_inlined.onnx`, inputs `img0`,`img1`, output `mid`.
- Slow-mo is fixed **2×** (model exported at timestep=0.5).
- No new npm deps; ort-web from CDN. `web/*.js` are ES modules (`web/package.json` `type:module`).
- Pure/testable logic (`assembleSlowmo`) lives in a `.js` module importable by node (no DOM).
- Browser pixels are RGBA8 `ImageData`. `interpolate` takes two same-size `ImageData`, returns `ImageData`.
- Commit after each task. Branch `phase4lite-slowmo` (already checked out).

```js
import { toInput, fromOutput } from './rife_prepost.js';

const EW = 1280, EH = 736; // engine (padded) input size

export async function createSession(ep) {
  ort.env.wasm.wasmPaths = 'https://cdn.jsdelivr.net/npm/onnxruntime-web/dist/';
  return await ort.InferenceSession.create('/assets/rife_lite_inlined.onnx',
    { executionProviders: [ep], graphOptimizationLevel: 'all' });
}

// Two same-size ImageData -> interpolated middle-frame ImageData.
export async function interpolate(session, a, b) {
  if (a.width !== b.width || a.height !== b.height) {
    throw new Error(`frame size mismatch: ${a.width}x${a.height} vs ${b.width}x${b.height}`);
  }
  const w = a.width, h = a.height;
  const feeds = {};
  feeds[session.inputNames[0]] = new ort.Tensor('float32', toInput(a.data, w, h, EW, EH), [1, 3, EH, EW]);
  feeds[session.inputNames[1]] = new ort.Tensor('float32', toInput(b.data, w, h, EW, EH), [1, 3, EH, EW]);
  const out = await session.run(feeds);
  const chw = out[session.outputNames[0]].data;
  return new ImageData(fromOutput(chw, w, h, EW, EH), w, h);
}
```

- [ ] **Step 3: Implement `web/slowmo.js`:**

```js
// Interleave frames with the mids computed between them:
// frames [F0..F_{n-1}], mids [M0..M_{n-2}] -> [F0,M0,F1,M1,...,F_{n-1}] (length 2n-1).
export function assembleSlowmo(frames, mids) {
  const out = [];
  for (let i = 0; i < frames.length; i++) {
    out.push(frames[i]);
    if (i < mids.length) out.push(mids[i]);
  }
  return out;
}
```

```html
<script type="module">
import { createSession, interpolate } from './rife_session.js';
import { assembleSlowmo } from './slowmo.js';

const MAXW = 1280, MAXH = 720;
const $ = id => document.getElementById(id);
const st = (m, bad) => { $('status').textContent += m + '\n'; if (bad) $('status').classList.add('bad'); };
const vid = $('vid'), out = $('out'), octx = out.getContext('2d');

let frames = [];   // sampled ImageData (native size)
let seq = [];      // assembled ImageData sequence to play
let sess = null, sessEp = null;

function loadVideo(src) {
  return new Promise((res, rej) => {
    vid.onloadeddata = () => res();
    vid.onerror = () => rej(new Error('video load/decode failed (codec unsupported?)'));
    vid.src = src;
  });
}
function seekTo(t) {
  return new Promise((res) => { vid.onseeked = () => res(); vid.currentTime = t; });
}
function grabFrame() {
  const s = Math.min(1, MAXW / vid.videoWidth, MAXH / vid.videoHeight);
  const w = Math.max(1, Math.round(vid.videoWidth * s)), h = Math.max(1, Math.round(vid.videoHeight * s));
  const off = new OffscreenCanvas(w, h), o = off.getContext('2d');
  o.drawImage(vid, 0, 0, w, h);
  return o.getImageData(0, 0, w, h);
}

async function sampleFrames(n, fps) {
  const got = [];
  for (let i = 0; i < n; i++) {
    await seekTo(i / fps);
    got.push(grabFrame());
    st(`sampled frame ${i + 1}/${n}`);
  }
  return got;
}

async function ensureSession(ep) {
  if (sess && sessEp === ep) return sess;
  st('creating session on ' + ep + ' ...');
  sess = await createSession(ep); sessEp = ep;
  st('session ready');
  return sess;
}

$('file').onchange = async e => {
  try { await loadVideo(URL.createObjectURL(e.target.files[0])); st('video loaded ' + vid.videoWidth + 'x' + vid.videoHeight); }
  catch (err) { st('ERROR: ' + err.message, true); }
};
$('demoBtn').onclick = async () => {
  try { await loadVideo('/demo/test_720p.mp4'); st('demo clip loaded ' + vid.videoWidth + 'x' + vid.videoHeight); }
  catch (err) { st('ERROR: ' + err.message, true); }
};

$('computeBtn').onclick = async () => {
  try {
    if (!vid.videoWidth) { st('load a clip first', true); return; }
    const ep = $('ep').value, n = +$('nFrames').value, fps = +$('fps').value;
    const s = await ensureSession(ep);
    frames = await sampleFrames(n, fps);
    const mids = [];
    const t0 = performance.now();
    for (let i = 0; i < frames.length - 1; i++) {
      st(`computing ${i + 1}/${frames.length - 1} ...`);
      mids.push(await interpolate(s, frames[i], frames[i + 1]));
    }
    const dt = performance.now() - t0;
    seq = assembleSlowmo(frames, mids);
    st(`done: ${frames.length} frames -> sequence ${seq.length} (2x), ${(dt / Math.max(1, mids.length)).toFixed(0)} ms/interp · ${ep}`);
    $('playBtn').disabled = false; $('pauseBtn').disabled = false;
    draw(0); startPlayback();
  } catch (err) { st('ERROR: ' + (err.stack || err.message || err), true); }
};

function draw(i) {
  const src = $('interp').checked ? seq : frames;
  if (!src.length) return;
  const f = src[i % src.length];
  out.width = f.width; out.height = f.height;
  octx.putImageData(f, 0, 0);
}

let raf = null, idx = 0, lastT = 0;
function startPlayback() {
  cancelAnimationFrame(raf); idx = 0; lastT = 0;
  const step = (ts) => {
    const src = $('interp').checked ? seq : frames;
    const interval = 1000 / (+$('fps').value);
    if (!lastT) lastT = ts;
    if (ts - lastT >= interval) {
      lastT = ts; idx++;
      if (idx >= src.length) { if ($('loop').checked) idx = 0; else { draw(src.length - 1); return; } }
      draw(idx);
    }
    raf = requestAnimationFrame(step);
  };
  raf = requestAnimationFrame(step);
}
$('playBtn').onclick = () => startPlayback();
$('pauseBtn').onclick = () => cancelAnimationFrame(raf);
</script>
</body>
</html>
```
